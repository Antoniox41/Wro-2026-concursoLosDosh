// =========================================================
//                      KALMAN 1D
// =========================================================
struct Kalman1D {
    float x;      //Estimación de la medida
    float P;      //varianza de la estimación
    float Q;      //varianza del ruido del modelo (incertidumbre al predecir)
    float R;      //varianza del ruido de la medición
};

float Pini = 200;  //varianza de la estimación previa inicial (se actualiza)
float Qini = 3;    //varianza del ruido del modelo (no se actualiza)
float Rlat = 50;   //varianza del ruido medición sensores laterales
float Rdel = 80;   //varianza del ruido medicion sensores delantero y trasero.

Kalman1D kL = {0, Pini, Qini, Rlat};
Kalman1D kF = {0, Pini, Qini, Rdel};
Kalman1D kR = {0, Pini, Qini, Rlat};
Kalman1D kB = {0, Pini, Qini, Rdel};

float kalmanUpdate(Kalman1D &k, float m) {
    k.P = k.P + k.Q;  //Estimación de la incertidumbre de la medida (incertidumbre previa + ruido modelo)
    float K = k.P / (k.P + k.R); //K es la gancia de Kalman 
    k.x = k.x + K * (m - k.x);   //estimación de magnitud medida 
    k.P = (1 - K) * k.P;   //actualización de la estimación de la incertidumbre
    return k.x;
}

float mide1(){    
  digitalWrite(trigPin1, LOW);
  delayMicroseconds(2);
  // Enviar pulso ultrasónico
  digitalWrite(trigPin1, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin1, LOW);
  // Leer el eco
  duracion = pulseIn(echoPin1, HIGH);
  // Calcular distancia en cm
  distancia = duracion * 0.034 / 2;
  return distancia;
}

void centraEnPista(int entRef){ 
mide33 = mide3();
if (mide33< 200 && mide33>0) {
  flag3 = 1; 
  mide33 = kalmanUpdate(kL, mide33);
  }//rechaza medidas largas
 mide11 = mide1();
if (mide11< 200 && mide11>0) {
  flag1 = 1;
  kalmanUpdate(kL, mide11);
  }//rechaza medidas largas
if (flag1 && flag3) { //si las medidas son válidas calcula la corrección del volante
  posicionTransversal = mide33-mide11;
  error = posicionTransversal - entRef;
  controlPropor = kp*error;
  volantePropor = map(controlPropor,-200,200,20,160);
  if (volantePropor < 20){volantePropor=20;}
  if (volantePropor > 160){volantePropor=160;}
  volante.write(volantePropor);
  flag1=0; flag3=0;
  delay(T);
  }
}

void esquivaObstaculo(){
  int areaRojo = 0;
  int areaVerde = 0;
  //Leer objetos
  while (huskylens.available()) {   //LeerObjeto
    HUSKYLENSResult result = huskylens.read();
    int area = result.width * result.height;

    if (result.ID == 1 && area > areaRojo) {
      areaRojo = area;
    }

    if (result.ID == 2 && area > areaVerde) {
      areaVerde = area;
    }
  }
  //Decidir cuál esquivar
 if (areaRojo > areaVerde && areaRojo > 500) {//"ROJO cercano → esquivar izquierda");
    volante.write(45); // izquierda
    analogWrite(inEn, velLen);
    delay(duracionEsquiva);
    volante.write(135); // vuelve al centro
    delay(duracionEsquiva);
    analogWrite(inEn, vel);

  } 
  else if (areaVerde > areaRojo && areaVerde > 500) {//VERDE cercano → esquivar derecha");
   volante.write(135); // izquierda
    analogWrite(inEn, velLen);
    delay(duracionEsquiva);
    volante.write(45); // vuelve al centro
    delay(duracionEsquiva);
    analogWrite(inEn, vel);
  } 
  
}
